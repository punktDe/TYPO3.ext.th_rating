348  public function getVote(string $prefixId, array $settings, Rating $rating): Vote
349  {
350  // initialize variables
352  $vote = null;
354  $voter = null;
355 
356  //first fetch real voter or anonymous
358  $frontendUserUid = $this->accessControllService->getFrontendUserUid();
359  if (!$frontendUserUid && !empty($settings['mapAnonymous'])) {
360  //set anonymous vote
361  $voter = $this->accessControllService->getFrontendVoter($settings['mapAnonymous']);
362  $anonymousRating = json_decode(
363  $_COOKIE[$prefixId . '_AnonymousRating_' . $rating->getUid()],
364  true,
365  512,
366  JSON_THROW_ON_ERROR
367  );
368  if (!empty($anonymousRating['voteUid'])) {
369  $vote = $this->voteRepository->findByUid($anonymousRating['voteUid']);
370  }
371  } elseif ($frontendUserUid) {
372  //set FEUser if one is logged on
373  $voter = $this->accessControllService->getFrontendVoter($frontendUserUid);
374  if ($voter instanceof \Thucke\ThRating\Domain\Model\Voter) {
375  $vote = $this->voteRepository->findMatchingRatingAndVoter($rating, $voter);
376  }
377  }
378  //voting not found in database or anonymous vote? - create new one
379  $voteValidator = $this->objectManager->get(VoteValidator::class);
380  if ($voteValidator->validate($vote)->hasErrors()) {
381  $vote = $this->objectManager->get(Vote::class);
382  $ratingValidator = $this->objectManager->get(RatingValidator::class);
383  if (!$ratingValidator->validate($rating)->hasErrors()) {
384  $vote->setRating($rating);
385  }
386  if ($voter instanceof \Thucke\ThRating\Domain\Model\Voter) {
387  $vote->setVoter($voter);
388  }
389  }
390 
391  return $vote;
