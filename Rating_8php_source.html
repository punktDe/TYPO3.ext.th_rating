276  public function checkCurrentrates()
277  {
278  $currentratesDecoded['weightedVotes'] = [];
279  $currentratesDecoded['sumWeightedVotes'] = [];
280  $numAllVotes = 0;
281  $numAllAnonymousVotes = 0;
282  foreach ($this->getRatingobject()->getStepconfs() as $stepConf) {
283  $stepOrder = $stepConf->getSteporder();
284  $voteCount = $this->voteRepository->countByMatchingRatingAndVote($this, $stepConf);
285  $anonymousCount = $this->voteRepository->countAnonymousByMatchingRatingAndVote(
286  $this,
287  $stepConf,
288  $this->settings['mapAnonymous']
289  );
290  $currentratesDecoded['weightedVotes'][$stepOrder] = $voteCount * $stepConf->getStepweight();
291  $currentratesDecoded['sumWeightedVotes'][$stepOrder] =
292  $currentratesDecoded['weightedVotes'][$stepOrder] * $stepOrder;
293  $numAllVotes += $voteCount;
294  $numAllAnonymousVotes += $anonymousCount;
295  }
296  $currentratesDecoded['numAllVotes'] = $numAllVotes;
297  $currentratesDecoded['anonymousVotes'] = $numAllAnonymousVotes;
298  $this->currentrates = json_encode($currentratesDecoded);
299  }

356  public function getCurrentrates(): array
357  {
358  $currentratesDecoded = json_decode($this->currentrates, true);
359  if (empty($currentratesDecoded['numAllVotes'])) {

361  $currentratesDecoded = json_decode($this->currentrates, true);

372
