271  public function checkCurrentrates()
272  {
273  $currentratesDecoded['weightedVotes'] = [];
274  $currentratesDecoded['sumWeightedVotes'] = [];
275  $numAllVotes = 0;
276  $numAllAnonymousVotes = 0;
277  foreach ($this->getRatingobject()->getStepconfs() as $stepConf) {
278  $stepOrder = $stepConf->getSteporder();
279  $voteCount = $this->voteRepository->countByMatchingRatingAndVote($this, $stepConf);
280  $anonymousCount = $this->voteRepository->countAnonymousByMatchingRatingAndVote(
281  $this,
282  $stepConf,
283  $this->settings['mapAnonymous']
284  );
285  $currentratesDecoded['weightedVotes'][$stepOrder] = $voteCount * $stepConf->getStepweight();
286  $currentratesDecoded['sumWeightedVotes'][$stepOrder] =
287  $currentratesDecoded['weightedVotes'][$stepOrder] * $stepOrder;
288  $numAllVotes += $voteCount;
289  $numAllAnonymousVotes += $anonymousCount;
290  }
291  $currentratesDecoded['numAllVotes'] = $numAllVotes;
292  $currentratesDecoded['anonymousVotes'] = $numAllAnonymousVotes;
293  $this->currentrates = $this->jsonService->encodeToJson($currentratesDecoded);
294  }

351  public function getCurrentrates(): array
352  {
353  $currentratesDecoded = $this->jsonService->decodeJsonToArray($this->currentrates);
354  if (empty($currentratesDecoded['numAllVotes'])) {
355  $this->checkCurrentrates();
356  $currentratesDecoded = $this->jsonService->decodeJsonToArray($this->currentrates);
357  }
358  $numAllVotes = $currentratesDecoded['numAllVotes'];
359  if (!empty($numAllVotes)) {
360  $currentrate = array_sum($currentratesDecoded['sumWeightedVotes']) / $numAllVotes;
361  } else {
362  $currentrate = 0;
363  $numAllVotes = 0;
364  }
365 
366  $sumAllWeightedVotes = array_sum($currentratesDecoded['weightedVotes']);
367 
368  //initialize array to handle missing stepconfs correctly
369  $currentPollDimensions = [];
370 
371  foreach ($this->getRatingobject()->getStepconfs() as $stepConf) {
372  if (empty($sumAllWeightedVotes)) {
373  //set current polling styles to zero percent and prevent division by zero error in lower formula
374  $currentPollDimensions[$stepConf->getStepOrder()]['pctValue'] = 0;
375  } else {
376  /* calculate current polling styles -> holds a percent value for usage in CSS
377  to display polling relations */
378  $currentPollDimensions[$stepConf->getStepOrder()]['pctValue'] =
379  round(
380  ($currentratesDecoded['weightedVotes'][$stepConf->getStepOrder()] * 100) /
381  $sumAllWeightedVotes,
382  1
383  );
384  }
385  }
386 
387  return ['currentrate' => $currentrate,
388  'weightedVotes' => $currentratesDecoded['weightedVotes'],
389  'sumWeightedVotes' => $currentratesDecoded['sumWeightedVotes'],
390  'anonymousVotes' => $currentratesDecoded['anonymousVotes'],
391  'currentPollDimensions' => $currentPollDimensions,
392  'numAllVotes' => $numAllVotes, ];
393  }
