262  public function checkCurrentrates()
263  {
264  $currentratesDecoded['weightedVotes'] = [];
265  $currentratesDecoded['sumWeightedVotes'] = [];
266  $numAllVotes = 0;
267  $numAllAnonymousVotes = 0;
268  foreach ($this->getRatingobject()->getStepconfs() as $stepConf) {
269  $stepOrder = $stepConf->getSteporder();
270  $voteCount = $this->voteRepository->countByMatchingRatingAndVote($this, $stepConf);
271  $anonymousCount = $this->voteRepository->countAnonymousByMatchingRatingAndVote(
272  $this,
273  $stepConf,
274  $this->settings['mapAnonymous']
275  );
276  $currentratesDecoded['weightedVotes'][$stepOrder] = $voteCount * $stepConf->getStepweight();
277  $currentratesDecoded['sumWeightedVotes'][$stepOrder] =
278  $currentratesDecoded['weightedVotes'][$stepOrder] * $stepOrder;
279  $numAllVotes += $voteCount;
280  $numAllAnonymousVotes += $anonymousCount;
281  }
282  $currentratesDecoded['numAllVotes'] = $numAllVotes;
283  $currentratesDecoded['anonymousVotes'] = $numAllAnonymousVotes;
284  $this->currentrates = json_encode($currentratesDecoded, JSON_THROW_ON_ERROR);
285  }

342  public function getCurrentrates(): array
343  {
344  $currentratesDecoded = json_decode($this->currentrates, true, 512, JSON_THROW_ON_ERROR);
345  if (empty($currentratesDecoded['numAllVotes'])) {
346  $this->checkCurrentrates();
347  $currentratesDecoded = json_decode($this->currentrates, true, 512, JSON_THROW_ON_ERROR);
348  }
349  $numAllVotes = $currentratesDecoded['numAllVotes'];
350  if (!empty($numAllVotes)) {
351  $currentrate = array_sum($currentratesDecoded['sumWeightedVotes']) / $numAllVotes;
352  } else {
353  $currentrate = 0;
354  $numAllVotes = 0;
355  }
356 
357  $sumAllWeightedVotes = array_sum($currentratesDecoded['weightedVotes']);
358 
359  //initialize array to handle missing stepconfs correctly
360  $currentPollDimensions = [];
361 
362  foreach ($this->getRatingobject()->getStepconfs() as $stepConf) {
363  if (empty($sumAllWeightedVotes)) {
364  //set current polling styles to zero percent and prevent division by zero error in lower formula
365  $currentPollDimensions[$stepConf->getStepOrder()]['pctValue'] = 0;
366  } else {
367  /* calculate current polling styles -> holds a percent value for usage in CSS
368  to display polling relations */
369  $currentPollDimensions[$stepConf->getStepOrder()]['pctValue'] =
370  round(
371  ($currentratesDecoded['weightedVotes'][$stepConf->getStepOrder()] * 100) /
372  $sumAllWeightedVotes,
373  1
374  );
375  }
376  }
377 
378  return ['currentrate' => $currentrate,
379  'weightedVotes' => $currentratesDecoded['weightedVotes'],
380  'sumWeightedVotes' => $currentratesDecoded['sumWeightedVotes'],
381  'anonymousVotes' => $currentratesDecoded['anonymousVotes'],
382  'currentPollDimensions' => $currentPollDimensions,
383  'numAllVotes' => $numAllVotes, ];
384  }
