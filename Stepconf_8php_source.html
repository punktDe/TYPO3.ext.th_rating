211  public function addStepname(Stepname $stepname): bool
212  {
213  $success = true;
214  $stepname->setStepconf($this);
215  if (!$this->stepnameRepository->existStepname($stepname)) {
216  $defaultLanguageObject = $this->stepnameRepository->findDefaultStepname($stepname);
217  if (is_object($defaultLanguageObject)) {
218  //handle localization if an entry for the default language exists
219  $stepname->setL18nParent($defaultLanguageObject->getUid());
220  }
221  $this->stepnameRepository->add($stepname);
222  $this->extensionHelperService->persistRepository(StepnameRepository::class, $stepname);
223  $stepname->getStepconf()->getStepname();
224  $this->extensionHelperService->persistRepository(StepconfRepository::class, $this);
225  $this->extensionHelperService->clearDynamicCssFile();
226  $this->stepname = $stepname;
227  } else {
228  //warning - existing stepname entry for a language will not be overwritten
229  $success = false;
230  }
